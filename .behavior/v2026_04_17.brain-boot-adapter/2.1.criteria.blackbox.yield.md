# blackbox criteria: brain-boot-adapter

## usecase.1 = init default config

given(a repo with linked roles)
  when(user runs `rhx init claude --bind --roles mechanic,architect`)
    then(default config is created at `.agent/.brain/claude/config/scope=default/`)
      sothat(team can commit and share the default config)
    then(CLAUDE.md contains briefs from specified roles)
      sothat(brains boot with role instructions in cacheable position)
    then(CLAUDE.md has header comment with generation info)
      sothat(user knows not to edit manually)
    then(settings.json contains hooks config)
      sothat(dynamic hooks still function)
    then(stdout shows config path created)
      sothat(user knows where config lives)

given(a repo with prior default config)
  when(user runs `rhx init claude --bind --roles mechanic,architect,keyrack`)
    then(default config is regenerated with new role set)
      sothat(config stays current with role changes)

## usecase.2 = enroll with default config

given(a repo with default config committed)
  when(user runs `rhx enroll claude`)
    then(brain spawns with `CLAUDE_CONFIG_DIR=.agent/.brain/claude/config/scope=default`)
      sothat(brain reads CLAUDE.md from default config)
    then(no new config is created)
      sothat(default config is reused as-is)

given(a repo with default config committed)
  when(second user runs `rhx enroll claude` in same repo)
    then(CLAUDE.md content is byte-identical across machines)
      sothat(determinism enables shared cache)
      note(cache hits are not directly testable; test precondition instead)

## usecase.3 = enroll with custom roles

given(a repo with linked roles mechanic, architect, driver, keyrack)
  when(user runs `rhx enroll claude --roles +architect,-driver`)
    then(role spec is parsed: base=all linked, add=[architect], remove=[driver])
      note(base set = all roles linked in .agent/repo=*/role=*/)
      note(+ prefix adds role, - prefix removes role)
    then(final role set = mechanic, architect, keyrack (driver removed))
    then(scoped config is created at `.agent/.brain/claude/config/scope=$hash/`)
      sothat(custom role set has its own config)
    then(CLAUDE.md contains briefs from computed role set)
      sothat(brain boots with custom instructions)
    then(brain spawns with `CLAUDE_CONFIG_DIR` set to scoped config)
      sothat(brain uses custom config)
    then(scoped config dir has .gitignore with explanation comment)
      sothat(custom configs are not committed)
    then(stdout shows scoped config path created)
      sothat(user knows where config lives)

given(two users enroll with same custom roles)
  when(hash of role set matches)
    then(both use same scoped config dir)
      sothat(no duplicate configs created)

## usecase.4 = upgrade regenerates config

given(a repo with default config)
  when(user runs `rhx upgrade --roles mechanic,architect`)
    then(role packages are upgraded)
    then(default config CLAUDE.md is regenerated)
      sothat(upgraded briefs are reflected)

given(a repo with scoped configs)
  when(user runs `rhx upgrade --roles mechanic,architect`)
    then(only default config is regenerated)
      sothat(custom scoped configs require explicit re-enrollment)

## usecase.5 = clone and go

given(a repo with default config committed)
  when(new team member clones repo)
    then(default config is present (CLAUDE.md, settings.json))
      note(.credentials.json symlink NOT committed — created at enroll time)
  when(new team member has authenticated with claude)
    then(`~/.claude/.credentials.json` exists on their machine)
  when(new team member runs `rhx enroll claude`)
    then(.credentials.json symlink is created to local credentials)
    then(brain spawns successfully)
      sothat(no setup required beyond clone + auth)

## usecase.6 = determinism enables cache benefit

given(CLAUDE.md generated with same roles in different order)
  when(output is compared)
    then(output is byte-identical)
      sothat(determinism enables shared cache)
      note(cache hits are not directly testable; test precondition instead)

given(CLAUDE.md generated on different machines with same roles)
  when(output is compared)
    then(output is byte-identical)
      sothat(cross-machine determinism enables shared cache)

## usecase.7 = boot order for cache optimization

given(roles from published packages (ehmpathy/mechanic) and local roles (repo=.this/any))
  when(CLAUDE.md is generated)
    then(published roles appear before local roles)
      sothat(stable content is prefix, volatile content is suffix)
      sothat(local changes only invalidate suffix, not entire cache)

## edgecase.1 = no roles linked

given(a repo with no roles linked)
  when(user runs `rhx init claude --bind`)
    then(error: no roles found in .agent/)
      sothat(user knows to link roles first)

## edgecase.2 = invalid brain

given(a repo with roles linked)
  when(user runs `rhx init foobar --bind --roles mechanic`)
    then(error: brain 'foobar' not supported)
      sothat(user knows valid brain options)

## edgecase.5 = invalid role slug

given(a repo with linked roles: mechanic, architect)
  when(user runs `rhx init claude --bind --roles foobar`)
    then(error: role 'foobar' not found in .agent/)
      sothat(user knows valid role options)

## edgecase.6 = scope hash collision

given(a scoped config exists at scope=$hash with roles=[mechanic])
  when(user runs `rhx enroll claude --roles +architect` and hash collides)
    then(error: scope hash collision with hint to delete scope or use different roles)
      sothat(user knows how to fix)

## edgecase.7 = invalid roles spec syntax

given(a repo with linked roles)
  when(user runs `rhx enroll claude --roles +-mechanic`)
    then(error: invalid role spec syntax)
      sothat(user knows correct +/- format)

## edgecase.3 = absent credentials

given(a repo with default config committed)
  when(new team member has NOT authenticated with claude)
  when(new team member runs `rhx enroll claude`)
    then(brain fails to start with auth error)
      sothat(user knows to authenticate first)

## edgecase.4 = absent default config

given(a repo with no default config)
  when(user runs `rhx enroll claude`)
    then(error: no config for scope=default. run 'rhx init claude --bind' first)
      sothat(user knows to run init first)
