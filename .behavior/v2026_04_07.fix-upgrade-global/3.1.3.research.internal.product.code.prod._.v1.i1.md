export const execNpmInstall = (
  input: { packages: string[] },
  context: ContextCli,
): void => {
  // detect package manager

## pattern 4: package manager detection

**location:** `src/domain.operations/upgrade/execNpmInstall.ts`

**what:** detects pnpm vs npm based on lock file presence.

**citation 4:**
```typescript
export const detectPackageManager = (input: {
  cwd: string;
}): 'pnpm' | 'npm' => {
  const hasPnpmLock = existsSync(join(input.cwd, 'pnpm-lock.yaml'));
  const hasNpmLock = existsSync(join(input.cwd, 'package-lock.json'));

  // prefer pnpm if its lock file exists
  if (hasPnpmLock) return 'pnpm';
```

**relation to wish:** global installs are npm-only (per vision). this detection is for local only.

**action:** [REUSE] — no change needed.
