# blackbox criteria: upgrade global rhachet by default

## episode: default upgrade via rhx

# usecase.1 = default upgrade (rhx invocation)

given(user invokes via rhx = global install detected)
  when(rhx upgrade)
    then(upgrades local rhachet)
      sothat(project has latest version)
    then(upgrades global rhachet)
      sothat(rhx command has latest version everywhere)
    then(upgrades roles)
    then(output shows both local and global upgraded)
      sothat(user knows what happened)

  when(rhx upgrade --which local)
    then(upgrades local rhachet only)
    then(does NOT upgrade global)
      sothat(user has explicit control)

  when(rhx upgrade --which global)
    then(upgrades global rhachet only)
    then(does NOT upgrade local)
      sothat(user has explicit control)

  when(rhx upgrade --which both)
    then(upgrades local rhachet)
    then(upgrades global rhachet)
      sothat(explicit is same as default)

# usecase.2 = npx invocation

given(user invokes via npx = no global install)
  when(npx rhachet upgrade)
    then(upgrades local rhachet only)
      sothat(no attempt to upgrade nonexistent global)
    then(output shows local upgraded)
    then(no error about global)
      sothat(npx users are not confused)

  when(npx rhachet upgrade --which global)
    then(fails with clear error)
      sothat(user knows global not available via npx)

# usecase.3 = global upgrade fails

given(user invokes via rhx)
  given(global npm prefix requires elevated permissions)
    when(rhx upgrade)
      then(upgrades local rhachet successfully)
      then(warns about global upgrade failure)
      then(surfaces hint for manual global upgrade)
        sothat(user knows how to fix)
      then(exits with success)
        sothat(local upgrade is not blocked by global failure)

# usecase.4 = already current

given(user invokes via rhx)
  given(global and local are already at latest version)
    when(rhx upgrade)
      then(reports "already current" for local)
      then(reports "already current" for global)
      then(no unnecessary network calls)
        sothat(upgrade is fast when already up to date)

# usecase.5 = version mismatch

given(user invokes via rhx)
  given(global is newer than local)
    when(rhx upgrade)
      then(upgrades local to latest)
      then(does NOT downgrade global)
        sothat(global stays at its version or upgrades)

given(user invokes via rhx)
  given(local is newer than global)
    when(rhx upgrade)
      then(upgrades global to latest)
      then(does NOT downgrade local)
        sothat(local stays at its version or upgrades)

## exchange: output format

# usecase.6 = output clarity

given(upgrade completes successfully)
  when(both local and global upgraded)
    then(output shows: rhachet upgraded (local))
    then(output shows: rhachet upgraded (global))
    then(output shows version transitions if changed)
      sothat(user sees exactly what changed)

given(upgrade completes successfully)
  when(only local upgraded)
    then(output shows: rhachet upgraded (local))
    then(no mention of global)
      sothat(output is not confusing)

given(global upgrade fails)
  when(local upgrade succeeds)
    then(output shows local success)
    then(output shows global warning with hint)
    then(hint suggests: sudo npm i -g rhachet@latest)
      sothat(user knows exactly what to do)
