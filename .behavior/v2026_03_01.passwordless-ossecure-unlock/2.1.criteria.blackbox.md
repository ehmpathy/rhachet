# blackbox criteria: passwordless os.secure unlock

---

# usecase.1 = unlock with prikey

## episode: init → set → unlock → get

given('keyrack initialized with custom owner and prikey')
  when('user sets a key with --env all')
    then('key is stored under $org.all.$key only')
      sothat('no duplicate entries are created')
    then('output shows single entry with correct tree structure')
  when('user unlocks with same prikey in new shell')
    then('unlock succeeds without prompts')
      sothat('CI/CD can unlock without human interaction')
    then('output shows unlocked keys')
  when('user runs echo $KEY')
    then('value is accessible in environment')

## exchange: unlock with prikey

given('keyrack with os.secure vault and custom prikey')
  when('rhachet keyrack unlock --owner $ORG --prikey $PATH --env all')
    then('exit code is 0')
    then('output contains "unlocked" message')
    then('no passphrase prompt appears')
  when('rhachet keyrack unlock --owner $ORG --prikey $WRONG_PATH --env all')
    then('exit code is non-zero')
    then('error message indicates prikey issue')
      sothat('user knows which key failed')

# usecase.2 = env=all stores once

## exchange: set with --env all

given('keyrack initialized')
  when('rhachet keyrack set $KEY --env all --value $VALUE')
    then('key stored under $org.all.$key')
    then('key NOT stored under $org.prod.$key')
    then('key NOT stored under $org.test.$key')
      sothat('no duplicates across environments')

## exchange: list with --env all keys

given('key set with --env all')
  when('rhachet keyrack list --owner $ORG')
    then('$org.all.$key appears once')
    then('$org.prod.$key does NOT appear')
    then('$org.test.$key does NOT appear')
      sothat('user sees deduplicated view')

## exchange: get recognizes env=all

given('key set with --env all')
  when('rhachet keyrack get $KEY --env prod')
    then('value is returned')
      sothat('env=all matches any environment')
  when('rhachet keyrack get $KEY --env test')
    then('value is returned')
  when('rhachet keyrack get $KEY --env all')
    then('value is returned')

# usecase.3 = tree output renders correctly

## exchange: set output structure

given('keyrack initialized')
  when('rhachet keyrack set $KEY --env all --value $VALUE')
    then('output tree has correct indentation')
    then('peer entries are aligned as peers')
    then('"done" appears at correct level')

expected structure:
```
🔐 keyrack set (org: $ORG, env: all)
   └─ $ORG.all.$KEY
      ├─ mech: PERMANENT_VIA_REPLICA
      └─ vault: os.secure
   └─ done
```

NOT:
```
🔐 keyrack set (org: $ORG, env: all)
   └─ $ORG.all.$KEY
      ├─ mech: ...
      └─ vault: ...
   └─ $ORG.prod.$KEY    ← wrong: duplicate
   └─ done              ← wrong: misaligned
```

# usecase.4 = passphrase fully deprecated (all vault adapters)

## exchange: --passphrase flag removed

given('any keyrack vault type')
  when('rhachet keyrack unlock --passphrase $PASS')
    then('error indicates --passphrase is not recognized')
      sothat('users migrate to --prikey')

## exchange: no interactive passphrase prompt from unlock flow

given('any keyrack vault type')
  when('rhachet keyrack unlock --owner $ORG (no --prikey, no default identity)')
    then('no interactive prompt appears')
    then('error indicates identity required')
      sothat('CI/CD never hangs on prompt')
note: vault adapters may invoke their own prompts internally — this criteria is for the top-level unlock flow

# usecase.5 = identity required (default or explicit)

## exchange: unlock with default identity

given('keyrack initialized with default identity')
  when('rhachet keyrack unlock --owner $ORG --env all (no --prikey)')
    then('exit code is 0')
    then('unlock succeeds with default identity')
      sothat('users with default ssh key dont need --prikey')

## exchange: unlock without any identity

given('keyrack with no default identity available')
  when('rhachet keyrack unlock --owner $ORG --env all (no --prikey)')
    then('exit code is non-zero')
    then('error message indicates identity required')
    then('no passphrase prompt appears')
      sothat('user knows exactly what is needed')

## exchange: set with default identity

given('keyrack initialized with default identity')
  when('rhachet keyrack set $KEY --value $VALUE (no --prikey)')
    then('exit code is 0')
    then('key encrypted with default identity')
      sothat('users with default ssh key dont need --prikey')

## exchange: set without any identity

given('keyrack with no default identity available')
  when('rhachet keyrack set $KEY --value $VALUE (no --prikey)')
    then('exit code is non-zero')
    then('error message indicates identity required')
      sothat('user knows identity is needed for encryption')
