# criteria.blackbox: aws sso locked vs absent

## usecase.1 = status check distinguishes absent from locked

given('key was never configured via keyrack')
  when('user checks status')
    then('status shows "absent 🫧"')
      sothat('user knows to run `keyrack set`')
    then('tip shows `keyrack set` command')
      sothat('user has clear next action')

given('key was configured via keyrack')
  given('key is not in daemon')
    when('user checks status')
      then('status shows "locked 🔒"')
        sothat('user knows the key exists but needs unlock')
      then('tip shows `keyrack unlock` command')
        sothat('user has clear next action')

given('key was configured via keyrack')
  given('key is in daemon')
    when('user checks status')
      then('status shows "granted 🔑"')
        sothat('user knows the key is ready to use')

## usecase.2 = keyrack set stocks inventory

given('key does not exist in inventory')
  when('user runs `keyrack set`')
    then('key is stored in vault')
    then('inventory entry is created')
      sothat('future status checks show locked instead of absent')

given('key already exists in inventory')
  when('user runs `keyrack set` again')
    then('key is updated in vault')
    then('inventory entry remains')
      sothat('idempotent operation does not corrupt inventory')

## usecase.3 = keyrack del removes inventory

given('key exists in inventory')
  when('user runs `keyrack del`')
    then('key is removed from vault')
    then('inventory entry is removed')
      sothat('future status checks show absent')

given('key does not exist in inventory')
  when('user runs `keyrack del`')
    then('operation succeeds')
      sothat('idempotent deletion is safe')

## usecase.4 = vault-agnostic inventory

given('key is set via any vault type')
  when('user runs `keyrack set --vault os.secure`')
    then('inventory entry is created')
  when('user runs `keyrack set --vault os.direct`')
    then('inventory entry is created')
  when('user runs `keyrack set --vault aws.config`')
    then('inventory entry is created')
  when('user runs `keyrack set --vault 1password`')
    then('inventory entry is created')
      sothat('all vaults benefit from locked vs absent distinction')

## usecase.5 = profile added outside keyrack

given('AWS profile exists in ~/.aws/config')
  given('profile was added via AWS CLI, not keyrack')
    when('user checks status for that key')
      then('status shows "absent 🫧"')
        sothat('keyrack only tracks keys it manages')
      then('tip shows `keyrack set` command')
        sothat('user can adopt the key under keyrack management')

## usecase.6 = orphan inventory recovery

given('inventory entry exists')
  given('vault entry was deleted externally')
    when('user checks status')
      then('status shows "locked 🔒"')
        sothat('user is guided to unlock rather than confused by absent')
    when('user runs `keyrack unlock`')
      then('unlock fails with clear error')
        sothat('user understands the key needs to be re-set')

## usecase.7 = no key→persistence leak

given('attacker has read access to ~/.rhachet/')
  when('attacker inspects inventory directory')
    then('attacker learns only "N keys were configured"')
    then('attacker cannot determine which keys or where they point')
      sothat('key→persistence lookup is never leaked')

given('user checks status')
  when('status is computed')
    then('no vault-specific lookup info is revealed')
    then('only "host knows: yes|no" is answered')
      sothat('security posture is maintained')
