## test infrastructure patterns

### withTempHome

**location:** `src/.test/infra/withTempHome.ts`

**[REUSE]** provides isolated HOME directory for integration tests.

```ts
export const withTempHome = (input: { name: string }): {
  path: string;
  setup: () => void;
  teardown: () => void;
} => {
  const tempHome = join(tmpdir(), `test-home-${input.name}-${Date.now()}`);
  return {
    path: tempHome,
    setup: () => {
      mkdirSync(tempHome, { recursive: true });
      process.env.HOME = tempHome;
    },
    teardown: () => {
      process.env.HOME = originalHome;
      rmSync(tempHome, { recursive: true, force: true });
    },
  };
};
```

**implications for blueprint:**
- reuse for mech adapter integration tests
- mech promptForSet tests need isolated HOME for .pem file reads
- vault adapter tests continue use of this pattern

### mockPromptHiddenInput

**location:** `src/.test/infra/mockPromptHiddenInput.ts`

**[REUSE]** queue-based mock for stdin prompts.

```ts
let mockPromptQueue: string[] = [];

export const setMockPromptValues = (values: string | string[]): void => {
  mockPromptQueue = Array.isArray(values) ? [...values] : [values];
};

export const genMockPromptHiddenInput = (): { promptHiddenInput: jest.Mock } => ({
  promptHiddenInput: jest.fn(async () => {
    if (mockPromptQueue.length === 0) {
      throw new UnexpectedCodePathError('no mock prompt values left');
    }
    return mockPromptQueue.shift();
  }),
});
```

**implications for blueprint:**
- reuse for mech adapter promptForSet tests
- queue-based design supports multi-step prompts (org → app → pem)
- mech tests will need more complex prompt sequences than vault tests

```ts
export const genMockVaultAdapter = (input?: {
  isUnlocked?: boolean;
  storage?: Record<string, string>;
}): KeyrackHostVaultAdapter => {
  const storage = input?.storage ?? {};
  let unlocked = input?.isUnlocked ?? false;
  return {
    unlock: jest.fn(async () => { unlocked = true; }),
    isUnlocked: jest.fn(async () => unlocked),
    get: jest.fn(async ({ slug }) => storage[slug] ?? null),
    set: jest.fn(async ({ slug, secret }) => { storage[slug] = secret; }),
    del: jest.fn(async ({ slug }) => { delete storage[slug]; }),
    relock: jest.fn(async () => { unlocked = false; }),
  };
};
```

**purpose:** test mech inference when vault supports multiple mechs

```ts
describe('inferMech', () => {
  given('[case1] vault supports single mech', () => {
    const mockVault = { supportedMechs: ['PERMANENT_VIA_REPLICA'] };

    when('[t0] inferMech is called', () => {

        const result = await inferMech({ vault: mockVault, key: 'API_KEY' });
        expect(result).toBe('PERMANENT_VIA_REPLICA');
      });
    });
  });

  given('[case2] vault supports multiple mechs', () => {
    const mockVault = { supportedMechs: ['EPHEMERAL_VIA_AWS_SSO', 'PERMANENT_VIA_AWS_KEY'] };

      });
    });

});
```
