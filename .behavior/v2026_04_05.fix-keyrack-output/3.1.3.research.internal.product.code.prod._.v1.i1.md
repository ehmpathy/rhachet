```ts
export const formatKeyrackGetOneOutput = (input: {
```

## pattern 4: formatKeyrackKeyBranch for tree structure [REUSE]

### what

pure function that formats a keyrack key entry as tree branch lines. handles granted, blocked, locked, absent, lost, and unlocked states.

### citations

[5] src/domain.operations/keyrack/cli/emitKeyrackKeyBranch.ts:7-77
```ts
export const formatKeyrackKeyBranch = (input: {
  entry: KeyrackKeyBranchEntry;
  isLast: boolean;
}): string[] => {
  const prefix = input.isLast ? '   └─' : '   ├─';
  const indent = input.isLast ? '      ' : '   │  ';
  const entry = input.entry;
  const lines: string[] = [];

  if (entry.type === 'granted') {
    lines.push(`${prefix} ${entry.grant.slug}`);
    lines.push(`${indent}├─ vault: ${entry.grant.source.vault}`);
    lines.push(`${indent}├─ mech: ${entry.grant.source.mech}`);
    lines.push(`${indent}└─ status: granted 🔑`);
    return lines;
  }
  // ... other status handlers
};
```

### how it relates to wish

- reuse for vibes output format
- unaffected by new output modes

### decision

[REUSE] — keep as-is

```ts
// check if all keys are granted (strict mode enabled by default)
const mode = input.mode ?? 'strict';
const keysNotGranted = keysForEnv.filter((k) => k.status !== 'granted');
if (mode === 'strict' && keysNotGranted.length > 0) {
  // get formatted output (same stdout as CLI) and forward it
  const formatted = spawnSync(RHX_BIN, args, {
    encoding: 'utf8',
  });
  const output = [
    formatted.stdout,
    formatted.stderr,
    '✋ some keys were not granted, yet are strictly required\n',
    '   └─ ask a human to set the keys, then try again\n',

  process.stderr.write(output);
```

[12] src/domain.operations/keyrack/sourceAllKeysIntoEnv.ts:105-113

```ts
// inject granted secrets into process.env
for (const key of keysForEnv) {
  if (key.status !== 'granted') continue;
  const envVarName = key.grant.slug.split('.').pop();
  if (envVarName && !process.env[envVarName]) {
    process.env[envVarName] = key.grant.key.secret;
  }
}
```

### how it relates to wish

- CLI `source` subcommand should match SDK semantics
- `--strict` (default): fail if any key not granted, exit 2
- `--lenient`: skip not-granted keys silently, exit 0
- SDK is reference for behavior parity

### decision

[REUSE] — SDK pattern is the reference; CLI mirrors it
