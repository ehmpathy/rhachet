### citations

[1] blackbox/.test/infra/invokeRhachetCliBinary.ts:15-35
```ts
export const invokeRhachetCliBinary = (input: {
  args: string[];
  cwd: string;
  env?: Record<string, string>;
}): { stdout: string; stderr: string; status: number } => {
  const result = spawnSync(
    'npx',
    ['tsx', join(__dirname, '../../../bin/run'), ...input.args],
    {
      cwd: input.cwd,
      env: { ...process.env, ...input.env },
      encoding: 'utf8',
    },
  );
  return {
    stdout: result.stdout ?? '',
    stderr: result.stderr ?? '',
    status: result.status ?? 1,
  };
};
```

### how it relates to wish

- reuse for testing `keyrack get --value` and `keyrack source` commands
- verify stdout, stderr, and exit codes match criteria
- subprocess invocation ensures black-box testing

## pattern 3: snapshot sanitization [REUSE]

### what

`asSnapshotSafe` strips machine-specific content from test output for stable snapshots.

### citations

[4] blackbox/.test/infra/invokeRhachetCliBinary.ts:45-65
```ts
export const asSnapshotSafe = (input: {
  stdout: string;
  stderr: string;
}): { stdout: string; stderr: string } => {
  const sanitize = (str: string) =>
    str
      .replace(/\/tmp\/rhachet-test-[a-f0-9-]+/g, '/tmp/rhachet-test-UUID')
      .replace(/pid: \d+/g, 'pid: PID')
      .replace(/expires in: \d+m/g, 'expires in: Nm');
  return {
    stdout: sanitize(input.stdout),
    stderr: sanitize(input.stderr),
  };
};
```

### how it relates to wish

- reuse for `--value` and `source` output snapshots
- ensures stable snapshots across runs
- may need extension for new output formats

### decision

[REUSE] — use extant sanitizer, extend if needed
