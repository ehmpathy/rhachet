## proposed test approach

add new test case to `blackbox/cli/rhx.acceptance.test.ts`:

```typescript
given('[case5] rhx invoked via symlink', () => {
  const scene = useBeforeAll(async () => {
    const repo = await genTestTempRepo({ fixture: 'with-skills' });
    // create symlink: {tempdir}/rhx-link -> bin/rhx
    const symlinkPath = join(repo.path, 'rhx-link');
    symlinkSync(RHX_BIN, symlinkPath);
    return { repo, symlinkPath };
  });

  when('[t0] symlink is invoked with say-hello', () => {
    const result = useBeforeAll(async () =>
      spawnSync(scene.symlinkPath, ['say-hello'], { cwd: scene.repo.path, encoding: 'utf-8' }),
    );

    then('exits with status 0', () => {
      expect(result.status).toEqual(0);
    });

    then('output matches direct invocation', () => {
      const direct = invokeRhachetCliBinary({ binary: 'rhx', args: ['say-hello'], cwd: scene.repo.path });
      expect(result.stdout).toEqual(direct.stdout);
    });
  });
});
```
