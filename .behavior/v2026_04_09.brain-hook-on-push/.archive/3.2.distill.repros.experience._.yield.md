## reproduction feasibility

### available test utilities
- `genTestTempDir` — create temp .comms/ directory
- `useBeforeAll` — test setup cache
- `createMockAdapter` — mock hook command invocation
- file system operations for message simulation

### required setup
1. create temp .comms/ directory
2. drop message file(s)
3. invoke poll function
4. assert on file state and hook invocations

### concrete test sketch

```typescript
given('[case1] message in .comms/radio/', () => {
  const scene = useBeforeAll(async () => {
    const commsDir = await genTestTempDir();
    const message = new BrainHookPushMessage({
      id: getUuid(),
      for: { topic: 'radio', roles: '+mechanic' },
      timestamp: new Date().toISOString(),
      payload: { content: 'hello' },
    });
    await writeFile(
      join(commsDir, 'radio', `${message.id}.json`),
      JSON.stringify(serialize(message)),
    );
    return { commsDir, message };
  });

  when('[t1] poll fires', () => {
    const hookInvocations = useThen('hook invoked', async () => {
      const invocations: HookInvocation[] = [];
      await pollTopic(
        { topic: 'radio', commsDir: scene.commsDir, cloneRoles: ['mechanic'] },
        { onHook: (inv) => invocations.push(inv) },
      );
      return invocations;
    });

    then('message matched', () => {
      expect(hookInvocations).toHaveLength(1);
    });

    then('payload delivered', () => {
      expect(hookInvocations[0].payload).toEqual({ content: 'hello' });
    });

    then('message deleted', async () => {
      const files = await readdir(join(scene.commsDir, 'radio'));
      expect(files).toHaveLength(0);
    });
  });
});
```
