## section 4: poll adapter [NEW]

### 4.1 poll adapter pattern

**proposed pattern:**
```typescript
export const pollCommsDirectory = async (
  input: { topics: string[]; hooks: OnPushHook[] },
  context: { log: LogMethods },
): Promise<{ processed: BrainHookPushMessage[] }> => {
  const processed: BrainHookPushMessage[] = [];
  for (const topic of input.topics) {
    const dir = `.agent/.comms/${topic}`;
    const files = await readdir(dir);
    for (const file of files) {
      if (!file.endsWith('.json')) continue;
      const content = await readFile(join(dir, file), 'utf-8');
      const message = BrainHookPushMessage.build(JSON.parse(content));
      const hooksMatched = input.hooks.filter(h => matchesFilter(h.filter, message));
      for (const hook of hooksMatched) {
        await invokeHook({ hook, message });
      }
      await unlink(join(dir, file)); // consume message
      processed.push(message);
    }
  }
  return { processed };
};
```

**relation to wish:** core poll mechanism for `.comms/` directory.

### 4.2 role filter parser [NEW]

**proposed pattern:**
```typescript
export const parseBrainHookPushFilter = (filter: string): BrainHookPushFilter => {
  const parts = filter.split(',').map(p => p.trim());
  const includes: string[] = [];
  const excludes: string[] = [];
  const exact: string[] = [];
  for (const part of parts) {
    if (part.startsWith('+')) includes.push(part.slice(1));
    else if (part.startsWith('-')) excludes.push(part.slice(1));
    else exact.push(part);
  }
  return { includes, excludes, exact };
};
```

**relation to wish:** parses `+mechanic,-foreman` syntax from wish.

### 4.3 role filter matcher [NEW]

**proposed pattern:**
```typescript
export const matchesBrainHookPushFilter = (
  input: { cloneRoles: string[]; filter: BrainHookPushFilter },
): boolean => {
  // exact match: clone has exactly these roles
  if (input.filter.exact.length > 0) {
    const sorted = [...input.cloneRoles].sort();
    const expected = [...input.filter.exact].sort();
    return JSON.stringify(sorted) === JSON.stringify(expected);
  }
  // inclusion check
  for (const role of input.filter.includes) {
    if (!input.cloneRoles.includes(role)) return false;
  }
  // exclusion check
  for (const role of input.filter.excludes) {
    if (input.cloneRoles.includes(role)) return false;
  }
  return true;
};
```

**relation to wish:** implements filter semantics from wish.

## section 5: onStop poll hook [NEW]

### 5.1 onStop poll hook pattern

**proposed pattern:**
```typescript
// hook command that runs onStop
// 1. checks stop_hook_active to prevent infinite loops
// 2. polls .comms/ directory
// 3. if message found, exit(2) to force continuation
// 4. otherwise exit(0) to allow stop

export const onStopPollHook = `
  if [ "$stop_hook_active" = "true" ]; then exit 0; fi
  result=$(poll-comms --topics slack,github,radio)
  if [ "$result" != "[]" ]; then exit 2; fi
  exit 0
`;
```

**relation to wish:** mechanism to wake brain on push via onStop hook.
