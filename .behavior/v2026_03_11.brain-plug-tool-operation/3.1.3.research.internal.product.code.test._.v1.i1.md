```ts
export const genMockedBrainAtom = (input?: {
  repo?: string;
  slug?: string;
  description?: string;
  content?: string;
}): BrainAtom =>
  new BrainAtom({
    // ...
    ask: async (askInput) => {
      const outputParsed = askInput.schema.output.parse({
        content: input?.content ?? '__mock_response__',
      });
      // ...
      return new BrainOutput<typeof outputParsed, 'atom'>({
        output: outputParsed,
        calls: null,  // <-- always null, no tool support
        metrics: genMockedBrainOutputMetrics(),
        episode,
        series: null,
      });
    },
  });
```

**extension needed:**
- add `calls?: { tools: BrainPlugToolInvocation[] }` input parameter
- return `calls` in BrainOutput when provided
- enable tests to simulate brain deferral to tools

**proposed shape:**
```ts
export const genMockedBrainAtom = (input?: {
  // ... current fields
  calls?: { tools: BrainPlugToolInvocation[] } | null;
}): BrainAtom =>
  new BrainAtom({
    ask: async (askInput) => {
      // if calls provided, simulate tool deferral
      if (input?.calls) {
        return new BrainOutput({
          output: null,  // null when deferring to tools
          calls: input.calls,
          // ...
        });
      }
      // ... current behavior
    },
  });
```

```ts
import { BrainPlugToolExecution } from '@src/domain.objects/BrainPlugToolExecution';
import { asIsoDuration } from 'iso-time';

/**
 * .what = generates a mocked BrainPlugToolExecution for tests
 * .why = reduces boilerplate in tests for tool execution flows
 */
export const genMockedBrainPlugToolExecution = <TInput = unknown, TOutput = unknown>(
  input?: {
    exid?: string;
    slug?: string;
    input?: TInput;
    output?: TOutput;
    signal?: 'success' | 'error:constraint' | 'error:malfunction';
    timeMs?: number;
  },
): BrainPlugToolExecution<TInput, TOutput> =>
  new BrainPlugToolExecution<TInput, TOutput>({
    exid: input?.exid ?? '__mock_exid__',
    slug: input?.slug ?? '__mock_tool__',
    input: input?.input ?? ({} as TInput),
    output: input?.output ?? ({} as TOutput),
    signal: input?.signal ?? 'success',
    metrics: {
      cost: {
        time: asIsoDuration({ milliseconds: input?.timeMs ?? 100 }),
      },
    },
  });
```

## new test file: BrainPlugToolExecution.test.ts [NEW]

**file:** `src/domain.objects/BrainPlugToolExecution.test.ts` (new file)

**proposed test cases:**
```ts
import { given, then, when } from 'test-fns';
import { BrainPlugToolExecution } from './BrainPlugToolExecution';
import { genMockedBrainPlugToolExecution } from '@src/.test.assets/genMockedBrainPlugToolExecution';

describe('BrainPlugToolExecution', () => {
  given('[case1] a successful tool execution', () => {
    when('[t0] execution is created with signal success', () => {
      then('signal is success', () => {
        const exec = genMockedBrainPlugToolExecution({ signal: 'success' });
        expect(exec.signal).toEqual('success');
      });

      then('metrics.cost.time is IsoDuration', () => {
        const exec = genMockedBrainPlugToolExecution({ timeMs: 150 });
        expect(exec.metrics.cost.time).toEqual('PT0.15S');
      });
    });
  });

  given('[case2] a constraint error execution', () => {
    when('[t0] execution is created with signal error:constraint', () => {
      then('signal is error:constraint', () => {
        const exec = genMockedBrainPlugToolExecution({ signal: 'error:constraint' });
        expect(exec.signal).toEqual('error:constraint');
      });
    });
  });

  given('[case3] a malfunction error execution', () => {
    when('[t0] execution is created with signal error:malfunction', () => {
      then('signal is error:malfunction', () => {
        const exec = genMockedBrainPlugToolExecution({ signal: 'error:malfunction' });
        expect(exec.signal).toEqual('error:malfunction');
      });
    });
  });
});
```
