# blackbox criteria: brain plug tool operation

---

# usecase.1 = single tool call with BrainAtom

given(a brain atom with a tool plugged)
  when(brain defers to the tool)
    then(caller receives tool invocation with exid, slug, input)
      sothat(caller knows what tool to execute and with what input)
    then(caller can execute the tool and measure time)
      sothat(caller captures execution metrics)
    then(caller can compose execution result with exid, slug, input, output, signal, metrics)
      sothat(result correlates to the original invocation)
  when(caller continues episode with execution result)
    then(brain receives the tool result and continues)
      sothat(brain can use the result to complete its task)
    then(brain returns final output)
      sothat(caller gets the answer they asked for)

---

# usecase.2 = parallel tool calls with BrainAtom

given(a brain atom with multiple tools plugged)
  when(brain defers to multiple tools)
    then(caller receives array of tool invocations)
      sothat(caller knows all tools to execute)
    then(caller can execute tools in parallel)
      sothat(execution is efficient)
  when(caller continues episode with array of executions)
    then(brain receives all tool results)
      sothat(brain can synthesize from multiple sources)
    then(brain returns final output)
      sothat(caller gets the combined answer)

---

# usecase.3 = agentic tool loop with BrainRepl

given(a brain repl with tools that have execute operations)
  when(caller invokes repl.act with prompt and tools)
    then(repl handles tool execution internally)
      sothat(caller does not manage tool loop)
    then(repl chains tool calls until task is complete)
      sothat(complex workflows are automated)
    then(repl returns final output with no tool calls)
      sothat(caller receives complete result)

---

# usecase.4 = tool execution succeeds

given(a tool invocation)
  when(tool executes successfully)
    then(execution has signal 'success')
      sothat(brain knows tool worked)
    then(execution has output with tool result)
      sothat(brain can use the data)
    then(execution has metrics with time)
      sothat(execution duration is observable)

---

# usecase.5 = tool execution fails with constraint error

given(a tool invocation)
  when(tool fails due to input or business constraint)
    then(execution has signal 'error:constraint')
      sothat(brain knows to try different approach)
    then(execution has output with error details)
      sothat(brain can reason about the failure)
    then(execution has metrics with time)
      sothat(failure duration is observable)

---

# usecase.6 = tool execution fails with malfunction

given(a tool invocation)
  when(tool breaks unexpectedly)
    then(execution has signal 'error:malfunction')
      sothat(brain knows tool is broken)
    then(execution has output with error details)
      sothat(brain can report the issue)
    then(execution has metrics with time)
      sothat(failure duration is observable)

---

# usecase.7 = partial tool execution with mixed results

given(multiple tool invocations)
  when(some tools succeed and some fail)
    then(caller receives executions for all tools)
      sothat(no results are lost)
    then(each execution has its own signal)
      sothat(brain sees which succeeded and which failed)
    then(brain can continue with partial information)
      sothat(workflow is not blocked by single failure)

---

# usecase.8 = type-safe tool execution

given(a typed tool invocation with input type TInput)
  when(caller composes execution result)
    then(execution input matches invocation input type)
      sothat(type safety flows through round trip)
    then(execution output can be typed as TOutput)
      sothat(caller and brain share type expectations)

---

# usecase.9 = episode continuation semantics

given(an episode with prior exchanges)
  when(caller continues with tool executions)
    then(brain sees prior context plus tool results)
      sothat(conversation state is preserved)
    then(brain can reference prior exchanges in response)
      sothat(tool results are understood in context)

---

# usecase.10 = prompt union semantics

given(a brain ask operation)
  when(caller passes string prompt)
    then(brain treats it as initial or follow-up prompt)
      sothat(conversation can start or continue)
  when(caller passes array of tool executions)
    then(brain treats it as tool result continuation)
      sothat(tool results are first-class prompt type)
