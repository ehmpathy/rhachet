```ts
/**
 * .what = finds a prikey that can decrypt the owner's host manifest
 * .why = prikeys are not ordered; we try each until one works
 */
export const getOnePrikeyForOwner = async (
  input: {
    owner: string;
    prikeys: { discovered: string[]; supplied: string[] };
  },
): Promise<{ prikey: string; hostContext: KeyrackHostContext } | null>
```

### acceptance

## phase 2: fillKeyrackKeys orchestrator

### tasks

- [ ] create `src/domain.operations/keyrack/fillKeyrackKeys.ts`
- [ ] create `src/.test/assets/genMockKeyrackRepoManifest.ts`

### read before start

- [ ] `.behavior/v2026_03_19.keyrack-fill/3.3.1.blueprint.product.v1.i1.md` — section "fillKeyrackKeys.ts orchestrator"
- [ ] `.behavior/v2026_03_19.keyrack-fill/3.1.3.research.internal.product.code.prod._.v1.i1.md` — patterns 2, 3, 4, 7, 8

### implementation

the orchestrator:
1. loads repo manifest
2. gets all keys for env via `getAllKeyrackSlugsForEnv`
3. for each key × each owner:
   - finds prikey via `getOnePrikeyForOwner`
   - checks if already set (skip if not --refresh)
   - infers vault if not prescribed
   - sets via `setKeyrackKey` (with `withStdoutPrefix` for nested output)
   - unlocks via `unlockKeyrackKeys`
   - verifies via `getKeyrackKeyGrant`
4. aggregates results and reports

### acceptance

```
given(repo manifest with 2 keys for env=test)
  given(owner=default has no keys set)
    when(fillKeyrackKeys called with env=test, owners=[default])
      then(prompts for each key via vault)
      then(sets, unlocks, gets each key)
      then(returns { results: [{ status: 'set' }, { status: 'set' }] })

given(repo manifest with 2 keys for env=test)
  given(owner=default has 1 key already set)
    when(fillKeyrackKeys called with env=test, owners=[default], refresh=false)
      then(skips already-set key with "already set, skip")
      then(prompts for absent key only)
      then(returns { results: [{ status: 'skipped' }, { status: 'set' }] })

given(repo manifest with 2 keys)
  when(fillKeyrackKeys called with key="SPECIFIC_KEY")
    then(filters to only that key)
    then(fails fast if key not found in manifest)

given(no prikey can decrypt owner's manifest)
  when(fillKeyrackKeys called)
    then(throws BadRequestError "no available prikey for owner=X")
```

### tasks

- [ ] update `src/contract/cli/invokeKeyrack.ts` — add `keyrack fill` subcommand

### read before start

- [ ] `.behavior/v2026_03_19.keyrack-fill/3.1.3.research.internal.product.code.prod._.v1.i1.md` — pattern 1 (CLI command structure)

### implementation

```ts
keyrack
  .command('fill')
  .description('fill keyrack keys from repo manifest')
  .requiredOption('--env <env>', 'environment to fill (test, prod, all)')
  .option('--owner <owner...>', 'owner(s) to fill (default: default)', ['default'])
  .option('--prikey <path...>', 'prikey(s) to consider for manifest decryption')
  .option('--key <key>', 'specific key to fill (default: all)')
  .option('--refresh', 'refresh even if already set')
  .action(async (opts) => {
    await fillKeyrackKeys({
      env: opts.env,
      owners: opts.owner,
      prikeys: opts.prikey ?? [],
      key: opts.key ?? null,
      refresh: opts.refresh ?? false,
    }, context);
  });
```

### acceptance

```
given(CLI invocation)
  when(rhx keyrack fill --env test)
    then(calls fillKeyrackKeys with env=test, owners=[default])

given(CLI invocation)
  when(rhx keyrack fill --env test --owner default --owner ehmpath)
    then(calls fillKeyrackKeys with owners=[default, ehmpath])

given(CLI invocation)
  when(rhx keyrack fill --env test --prikey ~/.ssh/ehmpath --prikey ~/.ssh/other)
    then(calls fillKeyrackKeys with prikeys=[~/.ssh/ehmpath, ~/.ssh/other])

given(CLI invocation without --env)
  then(exits with error "required option '--env' not specified")
```

### verification

```sh
npm run test:types
npm run build
rhx keyrack fill --help
```
