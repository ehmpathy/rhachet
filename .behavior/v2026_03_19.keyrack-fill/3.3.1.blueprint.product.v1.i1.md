# 3.3.1.blueprint.product.v1.i1

## summary

add `rhx keyrack fill` command that:
1. reads keys from repo manifest (keyrack.yml) for specified env
2. for each key × each owner: set → unlock → get (roundtrip verification)
3. skips already-configured keys (unless --refresh)
4. outputs tree-format progress with nested set output

this is pure orchestration over extant primitives. no new domain objects or DAOs needed.

### invokeKeyrack.ts update

add `keyrack fill` subcommand:

```ts
keyrack
  .command('fill')
  .description('fill keyrack keys from repo manifest')
  .requiredOption('--env <env>', 'environment to fill (test, prod, all)')
  .option('--owner <owner...>', 'owner(s) to fill (default: default)', ['default'])
  .option('--prikey <path...>', 'prikey(s) to consider for manifest decryption')
  .option('--key <key>', 'specific key to fill (default: all)')
  .option('--refresh', 'refresh even if already set')
  .action(async (opts) => {
    await fillKeyrackKeys({
      env: opts.env,
      owners: opts.owner,
      prikeys: opts.prikey ?? [],
      key: opts.key ?? null,
      refresh: opts.refresh ?? false,
    }, context);
  });
```

### getOnePrikeyForOwner.ts prikey finder

```ts
/**
 * .what = finds a prikey that can decrypt the owner's host manifest
 * .why = prikeys are not ordered; we try each until one works
 */
export const getOnePrikeyForOwner = async (
  input: {
    owner: string;
    prikeys: { discovered: string[]; supplied: string[] };
  },
): Promise<{ prikey: string | null; hostContext: KeyrackHostContext } | null> => {
  // 1. build prikey pool: discovered first, then supplied
  const prikeyPool = [...input.prikeys.discovered, ...input.prikeys.supplied];

  // 2. try each prikey until one works
  for (const prikey of prikeyPool) {
    const hostContext = await getKeyrackHostContext({ owner: input.owner, prikey });
    if (hostContext) {
      return { prikey, hostContext };
    }
  }

  // 3. none worked
  return null;
};
```

### fillKeyrackKeys.ts orchestrator

```ts
/**
 * .what = fills keyrack keys from repo manifest for specified owners
 * .why = eliminates adhoc fill commands; manifest becomes source of truth
 */
export const fillKeyrackKeys = async (
  input: {
    env: string;
    owners: string[];
    prikeys: string[];
    key: string | null;
    refresh: boolean;
  },
  context: { gitroot: string; log: LogMethods },
): Promise<{
  results: FillKeyResult[];
  summary: { set: number; skipped: number; failed: number };
}> => {
  // 1. load repo manifest
  const repoManifest = await daoKeyrackRepoManifest.get({ gitroot: context.gitroot });

  // 2. get all keys for env
  const allSlugs = getAllKeyrackSlugsForEnv({ manifest: repoManifest, env: input.env });

  // 3. filter to specific key if requested
  const slugs = input.key
    ? allSlugs.filter(s => s.includes(input.key))
    : allSlugs;

  if (slugs.length === 0) {
    if (input.key) {
      throw new BadRequestError(`key ${input.key} not found in manifest for env=${input.env}`);
    }
    context.log.warn('no keys found', { env: input.env });
    return { results: [], summary: { set: 0, skipped: 0, failed: 0 } };
  }

  // 4. emit header
  emitLine(`🔐 keyrack fill (env: ${input.env}, keys: ${slugs.length}, owners: ${input.owners.length})`);

  // 5. for each key
  const results: FillKeyResult[] = [];
  for (let i = 0; i < slugs.length; i++) {
    const slug = slugs[i];
    const keyName = asKeyrackKeyName({ slug });
    emitLine(`\n🔑 key ${i + 1}/${slugs.length}, ${keyName}, for ${input.owners.length} owners`);

    // 6. for each owner
    for (const owner of input.owners) {
      const isLastOwner = owner === input.owners[input.owners.length - 1];
      const ownerPrefix = isLastOwner ? '└─' : '├─';
      emitLine(`   ${ownerPrefix} for owner ${owner}`);

      // 7. find prikey that can decrypt this owner's host manifest
      const prikeysDiscovered = await getDiscoveredPrikeys();
      const prikeyResult = await getOnePrikeyForOwner({
        owner,
        prikeys: { discovered: prikeysDiscovered, supplied: input.prikeys },
      });

      if (!prikeyResult) {
        throw new BadRequestError(`no available prikey for owner=${owner}`);
      }

      const { prikey: prikeyFound, hostContext } = prikeyResult;

      // 8. check if already set
      const hostManifest = await daoKeyrackHostManifest.get({ owner, prikey: prikeyFound });
      const keyHost = hostManifest?.hosts[slug];
      const branchContinue = isLastOwner ? '   ' : '│  ';

      if (keyHost && !input.refresh) {
        emitLine(`   ${branchContinue}└─ ✓ already set, skip`);
        results.push({ slug, owner, status: 'skipped' });
        continue;
      }

      // 9. infer vault if not prescribed
      const keySpec = repoManifest.keys[slug];
      const vaultInferred = inferKeyrackVaultFromKey({ keyName });
      const vault = keySpec?.vault ?? vaultInferred ?? 'os.secure';
      const mechInferred = inferMechFromVault({ vault });
      const mech = keySpec?.mech ?? mechInferred ?? 'PERMANENT_VIA_REPLICA';

      // 10. emit "set the key" section with treebucket open
      const bucketIndent = `   ${branchContinue}│  │  `;
      emitLine(`   ${branchContinue}├─ set the key`);
      emitLine(`   ${branchContinue}│  ├─`);
      emitLine(`   ${branchContinue}│  │`);

      // 11. set key (stream nested output with prefix — interactive prompts flow through)
      await withStdoutPrefix(bucketIndent, async () => {
        await setKeyrackKey({
          key: keyName,
          env: input.env,
          org: repoManifest.org,
          vault,
          mech,
        }, hostContext);
      });

      // 12. close treebucket
      emitLine(`   ${branchContinue}│  │`);
      emitLine(`   ${branchContinue}│  └─`);

      // 13. emit "get after set, to verify" section
      emitLine(`   ${branchContinue}└─ get after set, to verify`);

      // 14. unlock key
      await unlockKeyrackKeys({
        env: input.env,
        keys: [keyName],
      }, await genContextKeyrackGrantUnlock({ owner, gitroot: context.gitroot, prikey: prikeyFound }));
      emitLine(`   ${branchContinue}   ├─ ✓ rhx keyrack unlock --key ${keyName} --env ${input.env} --owner ${owner}`);

      // 15. verify roundtrip
      const grant = await getKeyrackKeyGrant({
        key: keyName,
        env: input.env,
      }, await genContextKeyrackGrantGet({ owner, gitroot: context.gitroot }));

      if (!grant?.value) {
        emitLine(`   ${branchContinue}   └─ ✗ rhx keyrack get --key ${keyName} --env ${input.env} --owner ${owner}`);
        results.push({ slug, owner, status: 'failed' });
        continue;
      }

      emitLine(`   ${branchContinue}   └─ ✓ rhx keyrack get --key ${keyName} --env ${input.env} --owner ${owner}`);
      results.push({ slug, owner, status: 'set' });
    }
  }

  // 16. summary
  const summary = {
    set: results.filter(r => r.status === 'set').length,
    skipped: results.filter(r => r.status === 'skipped').length,
    failed: results.filter(r => r.status === 'failed').length,
  };

  emitLine(`\n🔐 keyrack fill complete (${summary.set + summary.skipped}/${slugs.length * input.owners.length} keys verified)`);

  return { results, summary };
};

type FillKeyResult = {
  slug: string;
  owner: string;
  status: 'set' | 'skipped' | 'failed';
};
```

## risks and mitigations

| risk | mitigation |
|------|------------|
| prikey resolution fails | fail-fast with clear error + hint |
| vault inference surprises | warn when vault inferred |
| partial success state | track per-key results, report which succeeded vs failed |
| roundtrip verification fails | clear error message, continue to next key |
