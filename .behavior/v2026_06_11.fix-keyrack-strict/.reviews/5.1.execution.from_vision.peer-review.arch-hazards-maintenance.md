  keys: Record<string, KeyrackKeySpec>;
  org: string;
}): Record<string, KeyrackKeySpec> => {
  const result: Record<string, KeyrackKeySpec> = {};
  for (const [originalSlug, spec] of Object.entries(input.keys)) {

      slug: newSlug,
    });
  }
  return result;
};
