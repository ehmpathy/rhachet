## pattern.3 = glob pattern matching via fast-glob

```ts
import fg from 'fast-glob';

const filterPathsByGlob = (input: { paths: string[]; glob: string }): string[] => {
  const pattern = input.glob;
  return fg.sync(pattern, {
    cwd: process.cwd(),
    onlyFiles: true,
    absolute: false,
  }).filter((p) => input.paths.includes(p));
};
```

**what:** filter paths via glob patterns using fast-glob.

## pattern.4 = recursive file collection

**citation:** `src/infra/filesystem/getAllFilesFromDir.ts:1-25`

```ts
export const getAllFilesFromDir = async (input: {
  dirPath: string;
  recursive?: boolean;
}): Promise<string[]> => {
  const entries = await fs.readdir(input.dirPath, { withFileTypes: true });
  const files: string[] = [];
  for (const entry of entries) {
    const fullPath = path.join(input.dirPath, entry.name);
    if (entry.isDirectory() && input.recursive !== false) {
      files.push(...(await getAllFilesFromDir({ dirPath: fullPath })));
    } else if (entry.isFile()) {
      files.push(fullPath);
    }
  }
  return files;
};
```

**what:** recursively collect all files from a directory.

**status:** [REUSE] — use to enumerate files in registered dirs before glob filtering.

## pattern.8 = path safety validation

**citation:** `src/infra/filesystem/assertPathWithinRepo.ts:1-15`

```ts
export const assertPathWithinRepo = (input: {
  path: string;
  repoRoot: string;
}): void => {
  const absolutePath = path.resolve(input.repoRoot, input.path);
  if (!absolutePath.startsWith(input.repoRoot)) {
    throw new BadRequestError('path must be within repo', {
      path: input.path,
      repoRoot: input.repoRoot,
    });
  }
};
```

**what:** fail-fast if path escapes repo root (security boundary).

**status:** [REUSE] — validate `--into` target is within repo.
