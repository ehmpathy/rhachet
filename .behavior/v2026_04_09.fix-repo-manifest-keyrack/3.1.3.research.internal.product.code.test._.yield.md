```ts
export const genTestTempRepo = (input: {
  fixture: TestRepoFixture;
}): { path: string } => {
  const tempDir = path.join(os.tmpdir(), `rhachet-test-${Date.now()}`);
  fs.cpSync(
    path.join(__dirname, '../assets', input.fixture),
    tempDir,
    { recursive: true }
  );

};
```

## pattern.2 = invokeRhachetCliBinary for subprocess invocation

**citation:** `accept.blackbox/.test/infra/invokeRhachetCliBinary.ts`

```ts
export const invokeRhachetCliBinary = (input: {
  args: string[];
  cwd: string;
}): { status: number; stdout: string; stderr: string } => {
  const result = spawnSync('npx', ['rhachet', ...input.args], {
    cwd: input.cwd,
    encoding: 'utf8',
  });
  return {
    status: result.status ?? 1,
    stdout: result.stdout,
    stderr: result.stderr,
  };
};
```

**what:** invoke CLI as subprocess, capture stdout/stderr/status.

**status:** [REUSE] — same pattern for `repo compile` acceptance tests.

## pattern.3 = BDD test structure

**citation:** `accept.blackbox/cli/roles.boot.acceptance.test.ts:1-30`

```ts
describe('rhachet roles boot', () => {
  given('[case1] repo with role', () => {
    const repo = useBeforeAll(async () =>
      genTestTempRepo({ fixture: 'with-roles' }),
    );

    when('[t0] boot is invoked', () => {
      const result = useBeforeAll(async () =>
        invokeRhachetCliBinary({
          args: ['roles', 'boot', '--repo', '.this', '--role', 'any'],
          cwd: repo.path,
        }),
      );

      then('exits with status 0', () => {
        expect(result.status).toEqual(0);
      });
    });
  });
});
```

**what:** given/when/then structure with useBeforeAll for shared setup.
