# blackbox criteria: `rhachet repo compile`

## usecase.1 = compile role artifacts

given(role package with `.agent/repo=.this/role=mechanic/`)
  given(role has briefs/, skills/, readme.md)
    when(user runs `npx rhachet repo compile --from src --into dist`)
      then(briefs `**/*.md` and `**/*.min` files are copied to dist/)
        sothat(knowledge artifacts are distributed)
      then(skills `**/*.sh` and `**/*.jsonc` files are copied to dist/)
        sothat(executable capabilities are distributed)
      then(skills `**/template/**` and `**/templates/**` are copied to dist/)
        sothat(skill templates are distributed)
      then(readme.md is copied to dist/)
        sothat(role documentation is distributed)
      then(boot.yml is copied to dist/ if present)
        sothat(boot config is distributed)
      then(keyrack.yml is copied to dist/ if present)
        sothat(credential requirements are distributed)
      then(relative directory structure is preserved)
        sothat(paths match source layout)

given(role package with multiple roles)
  when(user runs `npx rhachet repo compile --from src --into dist`)
    then(all roles are compiled)
      sothat(multi-role packages work)

## usecase.2 = default exclusions

given(role with `.test/` directory)
  when(user runs `npx rhachet repo compile --from src --into dist`)
    then(`.test/**` files are excluded)
      sothat(test fixtures are not distributed)

given(role with `__test_assets__/` directory)
  when(user runs `npx rhachet repo compile --from src --into dist`)
    then(`__test_*/**` files are excluded)
      sothat(test assets are not distributed)

given(role with `.route/`, `.scratch/`, `.behavior/` directories)
  when(user runs `npx rhachet repo compile --from src --into dist`)
    then(these directories are excluded)
      sothat(dev artifacts are not distributed)

given(role with `*.test.sh` files in skills/)
  when(user runs `npx rhachet repo compile --from src --into dist`)
    then(`*.test.*` files are excluded)
      sothat(test files are not distributed)

given(role with hidden directories like `.cache/`)
  when(user runs `npx rhachet repo compile --from src --into dist`)
    then(`.*` directories are excluded)
      sothat(hidden files are not distributed)

## usecase.3 = custom include

given(role with `data/*.json` files not in default globs)
  when(user runs `npx rhachet repo compile --from src --into dist --include 'data/**/*.json'`)
    then(data/*.json files are copied to dist/)
      sothat(custom artifacts can be included)

## usecase.4 = custom exclude

given(role with vendor/ directory)
  when(user runs `npx rhachet repo compile --from src --into dist --exclude 'vendor/**'`)
    then(vendor/ files are excluded)
      sothat(custom artifacts can be excluded)

## usecase.5 = mixed include/exclude

given(role with `skills/*.test.sh` and `inits/*.test.sh`)
  when(user runs `npx rhachet repo compile --from src --into dist --include 'skills/**/*.test.sh' --exclude 'inits/**/*.test.sh'`)
    then(skills/*.test.sh files are copied)
      sothat(include overrides default exclusion)
    then(inits/*.test.sh files are excluded)
      sothat(exclude is applied per user request)

## usecase.6 = preserve extant dist/ contents

given(dist/ contains tsc output like `index.js`)
  when(user runs `npx rhachet repo compile --from src --into dist`)
    then(extant dist/ files are preserved)
      sothat(tsc output survives)
    then(role artifacts are added to dist/)
      sothat(both tsc and role artifacts coexist)

## usecase.7 = prune empty directories

given(src/ has empty directories after exclusions are applied)
  when(user runs `npx rhachet repo compile --from src --into dist`)
    then(empty directories are not created in dist/)
      sothat(dist/ is clean like rsync --prune-empty-dirs)

## output.1 = success feedback

given(compile succeeds)
  when(command completes)
    then(stdout shows roles compiled)
      sothat(user knows what was done)
    then(stdout shows artifact counts per role)
      sothat(user can verify completeness)
    then(exit code is 0)
      sothat(ci pipelines pass)

## output.2 = error feedback

given(compile fails)
  when(command exits)
    then(stderr shows error message)
      sothat(user knows what went wrong)
    then(exit code is non-zero)
      sothat(ci pipelines fail)
