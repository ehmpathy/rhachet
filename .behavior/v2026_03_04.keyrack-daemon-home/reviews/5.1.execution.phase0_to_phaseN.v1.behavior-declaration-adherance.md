  const realPath = realpathSync(homePath);               // ✅ expand symlinks

### 3. scheduleAutoTermination.ts (NEW)

**blueprint spec:**
```ts
scheduleAutoTermination
├── [+] scheduleAutoTermination(keyStore, interval = 15min)
│   ├── setInterval to check keyStore.entries().length === 0
│   └── if empty after initial population → graceful shutdown
```

**implementation:**
```ts
export const scheduleAutoTermination = (input: {
  keyStore: DaemonKeyStore;
  checkIntervalMs?: number;
}): NodeJS.Timeout => {
  const { keyStore, checkIntervalMs = 15 * 60 * 1000 } = input;  // ✅ 15min default
  let hasEverHadKeys = false;

  return setInterval(() => {
    const entries = keyStore.entries();  // ✅ check entries
    if (entries.length > 0) hasEverHadKeys = true;  // ✅ track if ever had keys
    if (hasEverHadKeys && entries.length === 0) {   // ✅ only if had keys before
      console.log('[keyrack-daemon] all keys expired, terminate');
      process.exit(0);  // ✅ graceful shutdown
    }
  }, checkIntervalMs);
};
```

**verdict: ✅ ADHERES** — matches blueprint exactly
